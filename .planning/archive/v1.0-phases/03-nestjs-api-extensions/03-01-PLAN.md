```typescript
@Controller('products')
export class ProductsController {
  constructor(private readonly productsService: ProductsService) {}

  @Get(':id/preview')
  async preview(@Param('id') id: string) {
    const product = await this.productsService.findOne(id);
    if (!product) throw new NotFoundException('Product not found');
    const rawData = (product.rawData as Record<string, unknown>) || {};
    return {
      template: product.processedData ? 'bold-vertical' : null,
      data: product.processedData || rawData,
      images: rawData['images'] || [],
    };
  }
}
```

From apps/server/src/agent-tasks/agent-tasks.service.ts (reference for pg_notify pattern):
```typescript
async create(agentType: string, input?: Record<string, unknown>) {
  const task = await this.prisma.agentTask.create({
    data: {
      agentType,
      ...(input && { input: input as any }),
    },
  });
  await this.prisma.$executeRawUnsafe(
    `SELECT pg_notify('new_agent_task', $1)`,
    task.id,
  );
  return task;
}
```
