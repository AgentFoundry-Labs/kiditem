    ```typescript
    const KST_OFFSET_MS = 9 * 60 * 60 * 1000;

    /**
     * Converts a JS Date to the UTC timestamp representing midnight KST (UTC+9).
     * Docker containers run UTC — this helper ensures Korean date bucketing is correct.
     *
     * Example: 2026-03-23T05:19:48Z (KST 14:19:48) → 2026-03-22T15:00:00.000Z (KST midnight)
     */
    export function kstDayStart(date: Date): Date {
      const kstMs = date.getTime() + KST_OFFSET_MS;
      const kstMidnightMs = Math.floor(kstMs / 86400000) * 86400000;
      return new Date(kstMidnightMs - KST_OFFSET_MS);
    }
    ```

    4. Create `apps/server/src/coupang-dashboard/coupang-dashboard.service.ts`:
    ```typescript
    import { Injectable } from '@nestjs/common';
    import { PrismaService } from '../prisma/prisma.service';
    import { kstDayStart } from '../common/kst';
    import { ORDER_STATUSES, RETURN_STATUSES } from '../coupang/constants';

    @Injectable()
    export class CoupangDashboardService {
      constructor(private readonly prisma: PrismaService) {}

      async getSummary(companyId: string) {
        const todayStart = kstDayStart(new Date());
        const todayEnd = new Date(todayStart.getTime() + 86400000);

        const [todayOrders, pendingAccept, pendingReturns] = await Promise.all([
          this.prisma.coupangOrder.aggregate({
            _sum: { totalPrice: true },
            _count: true,
            where: {
              companyId,
              orderedAt: { gte: todayStart, lt: todayEnd },
            },
          }),
          this.prisma.coupangOrder.count({
            where: { companyId, status: ORDER_STATUSES.ACCEPT },
          }),
          this.prisma.coupangReturn.count({
            where: { companyId, receiptStatus: RETURN_STATUSES.UC },
          }),
        ]);

        return {
          todayOrders: {
            count: todayOrders._count,
            revenue: todayOrders._sum.totalPrice ?? 0,
          },
          pendingAccept,
          pendingReturns,
        };
      }
    }
    ```
    CRITICAL constraints:
    - Use `Promise.all()` for all aggregation queries — no sequential `await` chains (STATE.md decision).
    - Use `kstDayStart()` for date range — NOT `new Date(y, m, d)` (that's the DashboardService anti-pattern).
    - Use `ORDER_STATUSES.ACCEPT` and `RETURN_STATUSES.UC` — NOT string literals.
    - Include `companyId` in every `where` clause (multi-tenant safety).
    - Dashboard reads from DB only — do NOT import or call any Coupang API functions.
    - Do NOT inject other domain services — PrismaService only (CLAUDE.md rule).
